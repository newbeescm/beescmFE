## Install
```shell
npm install beescm-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import Element from 'beescm-ui'

Vue.use(Element)

// or
import {
  Select,
  Button
  // ...
} from 'beescm-ui'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
